/// A page-locked (pinned) host memory buffer for faster PCIe transfers.
pub struct PinnedBuffer {
    ptr: *mut c_void,
    bytes: usize,
}

unsafe impl Send for PinnedBuffer {}
unsafe impl Sync for PinnedBuffer {}

impl PinnedBuffer {
    pub fn new(bytes: usize) -> Result<Self> {
        let mut ptr = std::ptr::null_mut();
        let res = unsafe { hipHostMalloc(&mut ptr, bytes, 0) };
        if res != hipSuccess || ptr.is_null() {
            return Err(Error::Backend(format!("hipHostMalloc failed: {}", res)));
        }
        Ok(Self { ptr, bytes })
    }

    pub fn as_mut_slice<T>(&mut self) -> &mut [T] {
        unsafe { std::slice::from_raw_parts_mut(self.ptr as *mut T, self.bytes / std::mem::size_of::<T>()) }
    }

    pub fn as_slice<T>(&self) -> &[T] {
        unsafe { std::slice::from_raw_parts(self.ptr as *const T, self.bytes / std::mem::size_of::<T>()) }
    }

    pub fn as_ptr(&self) -> *const c_void {
        self.ptr
    }

    pub fn as_mut_ptr(&self) -> *mut c_void {
        self.ptr
    }
}

impl Drop for PinnedBuffer {
    fn drop(&mut self) {
        unsafe {
            let _ = hipHostFree(self.ptr);
        }
    }
}

thread_local! {
    static PINNED_STAGING_POOL: std::cell::RefCell<Vec<PinnedBuffer>> = std::cell::RefCell::new(Vec::new());
}

pub fn get_pinned_buffer(bytes: usize) -> Result<PinnedBuffer> {
    let buf_opt = PINNED_STAGING_POOL.with(|pool| {
        let mut pool = pool.borrow_mut();
        if let Some(pos) = pool.iter().position(|b| b.bytes >= bytes) {
            Some(pool.remove(pos))
        } else {
            None
        }
    });
    if let Some(buf) = buf_opt {
        Ok(buf)
    } else {
        PinnedBuffer::new(bytes)
    }
}

pub fn release_pinned_buffer(buf: PinnedBuffer) {
    PINNED_STAGING_POOL.with(|pool| {
        let mut pool = pool.borrow_mut();
        if pool.len() < 8 {
            pool.push(buf);
        }
    });
}

    fn cpu_qkv_attention_ref(
        q: &[f32],
        k: &[f32],
        v: &[f32],
        seq_len: usize,
        num_heads: usize,
        num_kv_heads: usize,
        head_dim: usize,
    ) -> Vec<f32> {
        let mut out = vec![0.0f32; seq_len * num_heads * head_dim];
        let kv_mul = num_heads / num_kv_heads;

        for t in 0..seq_len {
            for h in 0..num_heads {
                let kv_h = h / kv_mul;
                let q_offset = t * num_heads * head_dim + h * head_dim;

                let mut scores = vec![0.0f32; t + 1];
                let mut max_score = -f32::INFINITY;
                for j in 0..=t {
                    let k_offset = j * num_kv_heads * head_dim + kv_h * head_dim;
                    let mut dot = 0.0f32;
                    for d in 0..head_dim {
                        dot += q[q_offset + d] * k[k_offset + d];
                    }
                    let score = dot / (head_dim as f32).sqrt();
                    scores[j] = score;
                    if score > max_score {
                        max_score = score;
                    }
                }

                let mut sum_exp = 0.0f32;
                for j in 0..=t {
                    scores[j] = (scores[j] - max_score).exp();
                    sum_exp += scores[j];
                }
                for j in 0..=t {
                    scores[j] /= sum_exp;
                }

                let out_offset = t * num_heads * head_dim + h * head_dim;
                for d in 0..head_dim {
                    let mut sum_val = 0.0f32;
                    for j in 0..=t {
                        let v_offset = j * num_kv_heads * head_dim + kv_h * head_dim;
                        sum_val += scores[j] * v[v_offset + d];
                    }
                    out[out_offset + d] = sum_val;
                }
            }
        }
        out
    }
